399 # Helper function to calculate invitation count server-side
400 defp calculate_invitation_count(selected_suggestions, manual_emails) do
401 
:-(
 selected_count = length(selected_suggestions || [])
402
403 
:-(
 email_count =
404 case manual_emails do
405 
:-(
 nil -> 0
406 
:-(
 "" -> 0
407 emails ->
408 emails
409 |> String.split(~r/[,\n]/)
410 |> Enum.map(&String.trim/1)
411 
:-(
 |> Enum.reject(&(&1 == ""))
412 
:-(
 |> length()
413 end
414
415 
:-(
 selected_count + email_count
416 end
