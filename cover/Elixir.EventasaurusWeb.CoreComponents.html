827 def translate_error({msg, opts}) do
828 # When using gettext, we typically pass the strings we want
829 # to translate as a static argument:
830 #
831 # # Translate the number of files with plural rules
832 # dngettext("errors", "1 file", "%{count} files", count)
833 #
834 # However the error messages in our forms and APIs are generated
835 # dynamically, so we need to translate them by calling Gettext
836 # with our gettext backend as first argument. Translations are
837 # available in the errors.po file (as we use the "errors" domain).
838 9 if count = opts[:count] do
839 
:-(
 Gettext.dngettext(EventasaurusWeb.Gettext, "errors", msg, msg, count, opts)
840 else
841 9 Gettext.dgettext(EventasaurusWeb.Gettext, "errors", msg, opts)
842 end
843 end
