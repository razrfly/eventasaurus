687 defp parse_signature_header(signature_header) do
688 # Stripe signature header format: "t=timestamp,v1=signature"
689 parts = String.split(signature_header, ",")
690
691 timestamp =
692 parts
693 |> Enum.find(&String.starts_with?(&1, "t="))
694 |> case do
695 "t=" <> timestamp -> timestamp
696 _ -> nil
697 end
698
699 signature =
700 parts
701 |> Enum.find(&String.starts_with?(&1, "v1="))
702 |> case do
703 "v1=" <> signature -> signature
704 _ -> nil
705 end
706
707 if timestamp && signature do
708 {:ok, timestamp, signature}
709 else
710 {:error, "Invalid signature header format"}
711 end
712 end
